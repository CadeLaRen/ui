  <script type="text/javascript">
    $('#drawers').jqfDrawers();
    $(document).ready(function(){
      $.jqf.waitStart({
        content:'Opening first drawer in 3 seconds...',
        object:$('#drawers'),
        onStop:function(){
          $.jqf.drawersOpen([$('#drawers-prize')]);
        },
        seconds:3
      });
    });
  </script>
